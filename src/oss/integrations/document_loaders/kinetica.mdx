from langchain_community.document_loaders.kinetica_loader import KineticaLoader# The following `QUERY` is an example which will not run; this# needs to be substituted with a valid `QUERY` that will return# data and the `SCHEMA.TABLE` combination must exist in Kinetica.QUERY = "select text, survey_id as source from SCHEMA.TABLE limit 10"kl = KineticaLoader(    query=QUERY,    host=HOST,    username=USERNAME,    password=PASSWORD,    metadata_columns=["source"],)kinetica_documents = kl.load()print(kinetica_documents)
```
